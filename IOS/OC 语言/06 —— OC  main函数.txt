---------------------------------------------------------------------------------------------------

#import <Foundation/Foundation.h> //表示包含Foundation框架中的Foundation.h文件


int main(int argc, const char * argv[]){
     @autoreleasepool { //在Java中，有辣鸡回收机制，
         NSLog(@"Hello, World!"); //向控制台输出信息。
     }

     return 0; 
}


@autoreleasepool{};
 在Java中，有辣鸡回收机制，系统会自动回收不再使用的对象；
 而OC是不支持辣鸡回收的，需要开发人员编写代码来释放对象所占的内存，与内存管理相关
